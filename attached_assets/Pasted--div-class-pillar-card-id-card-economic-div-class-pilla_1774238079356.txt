    function updateTime() {
        const now = new Date();
        const est = new Date(now.toLocaleString("en-US", {timeZone: "America/New_York"}));
        document.getElementById('datetime').textContent = est.toLocaleString('en-US', {
            weekday: 'short', month: 'short', day: 'numeric',
            hour: '2-digit', minute: '2-digit', second: '2-digit',
            timeZoneName: 'short'
        });
    }
    setInterval(updateTime, 1000);
    updateTime();

    function setGauge(gaugeValue) {
        const rotation = (gaugeValue / 100) * 180 - 90;
        document.getElementById('gauge-needle').setAttribute('transform', `rotate(${rotation}, 130, 140)`);
        document.getElementById('gauge-score-text').textContent = gaugeValue;
    }

    function getScoreClass(score) {
        if (score > 0.3) return 'bullish';
        if (score < -0.3) return 'bearish';
        return 'neutral';
    }

    function getBadgeClass(score) {
        if (score > 0.3) return 'score-bullish';
        if (score < -0.3) return 'score-bearish';
        return 'score-neutral';
    }

    function renderEconomic(data) {
        if (!data) return '<div class="stale-warning">⚠️ Data unavailable</div>';
        let html = '';
        if (data.status === 'stale') html += '<div class="stale-warning">⚠️ Stale data — last known values shown</div>';
        const events = data.events || [];
        if (events.length === 0) return html + '<div class="data-row"><span class="data-label">No red folder events today</span></div>';
        events.forEach(event => {
            const impactClass = event.market_impact === 'bullish' ? 'value-bullish' : event.market_impact === 'bearish' ? 'value-bearish' : 'value-neutral';
            html += `
                <div class="flag-item">
                    <div class="flag-title">${event.title}</div>
                    <div class="flag-meta">
                        <span class="flag-status status-${event.result === 'pending' ? 'awaiting' : 'resolved'}">${event.result === 'pending' ? 'Pending' : event.result.toUpperCase()}</span>
                        <span class="flag-source">${event.time_est} EST</span>
                    </div>
                    <div class="data-row">
                        <span class="data-label">Expected</span>
                        <span class="data-value">${event.forecast}</span>
                    </div>
                    <div class="data-row">
                        <span class="data-label">Actual</span>
                        <span class="data-value ${impactClass}">${event.actual}</span>
                    </div>
                    ${event.result !== 'pending' ? `<div class="data-row"><span class="data-label">Market Impact</span><span class="data-value ${impactClass}">${event.market_impact?.toUpperCase()}</span></div>` : ''}
                </div>`;
        });
        (data.warnings || []).forEach(w => {
            html += `<div class="stale-warning">${w}</div>`;
        });
        return html;
    }
